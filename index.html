1.1.1 Find the first n Ulam numbers where n is a positive integer.

//Limit
let limit = 13000;

//Initialize array for Ulam numbers
let arr = [];

//Compute Ulam numbers
function ulam() {
    arr.push(1);
    arr.push(2);

    for (let i = 3; i < limit; i++) {
        let count = 0;

        for (let j = 0; j < arr.length - 1; j++) {
            for (let k = j + 1; k < arr.length; k++) {
                if (arr[j] + arr[k] == i) {
                    count++;
                }
                if (count > 1) break;
            }
            if (count > 1) break;
        }

        if (count == 1) {
            arr.push(i);
        }
    }
}

ulam();

//Compute the first n Ulam numbers
function output(n) {
    for (x = 0; x < n; x++) {
        console.log(arr[x]);
    }
}

//Output the first 1000 Ulam numbers
output(1000);
